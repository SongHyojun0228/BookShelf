사람들이 많은 담은 순으로
**방법 : DB view**
- myLibrary 테이블에 몇 번 등장했는 지 미리 계산해주는 뷰 생성
CREATE OR REPLACE VIEW public.popular_books AS
SELECT 
    b.book_id,
    b.title,
    b.author,
    b.img,
    COUNT(m.mylibrary_id) as shelf_count -- MyLibrary에 몇 개 담겼는지 카운트
FROM 
    public."Books" b
LEFT JOIN 
    public."MyLibrary" m ON b.book_id = m.book_id
GROUP BY 
    b.book_id
ORDER BY 
    shelf_count DESC; -- 많이 담긴 순 정렬
