3. 검색 방식 분리 ( 새로운 검색 vs 더 불러오기 )
- handleSearch : 처음 검색 버튼 눌렀을 때 1페이지만 
- **"바닥에 닿았을 때 다음 페이지를 불러오는 함수"**를 하나 추가 
// 새로운 책 검색 (1페이지부터 다시 시작)
const handleSearch = async (e: React.FormEvent) => {
    e.preventDefault();
    if (!keyword.trim()) return;
    
    setPage(1); // 검색 시 무조건 1페이지로 초기화
    fetchBooks(1, true); // true = 기존 목록을 지우고 새로 덮어쓰기
}

// 실제로 API를 찔러서 데이터를 가져오는 공통 함수
const fetchBooks = async (pageNum: number, isNewSearch: boolean = false) => {
    setIsLoading(true);
    try {
        const ttbKey = '본인키';
        // start=${pageNum} 으로 변경된 것을 확인하세요! (MaxResults는 30)
        const url = `/aladin-api/ttb/api/ItemSearch.aspx?ttbkey=${ttbKey}&Query=${keyword}&QueryType=Keyword&MaxResults=30&start=${pageNum}&SearchTarget=Book&output=js&Version=20131101`;
        
        const response = await fetch(url);
        const data = await response.json();
        
        if(data.item) {
            if(isNewSearch) {
                // 아예 새로운 검색이면 덮어쓰기
                setSearchResults(data.item);
            } else {
                // 더보기(무한스크롤)면 기존 배열(prev) 뒤에 새 배열(data.item) 합치기
                setSearchResults(prev => [...prev, ...data.item]);
            }
        }
    } catch (e) {
        console.log(e);
    } finally {
        setIsLoading(false);
    }
}


4. 바닥 감지 센서(**ref**) 달기
- **useInView** 훅 사용
`import { useInView } from 'react-intersection-observer';`

function Searchpage() {
    // ... 기존 state 들 ...
    
    // 센서를 만듭니다. (ref: 센서 위치, inView: 화면에 센서가 보이면 true가 됨)
    const { ref, inView } = useInView();

    // inView가 true로 변하면(바닥에 닿으면) page 1 증가시키기
    useEffect(() => {
        if (inView && !isLoading && searchResults.length > 0) {
            setPage(prev => prev + 1);
        }
    }, [inView, isLoading]);
    
    // page가 변할 때마다 그 페이지 번호로 새 책들 밑에 이어붙이기 추가 요청
    useEffect(() => {
        if (page > 1) {
            fetchBooks(page, false); // false = 기존 목록 유지하면서 이어붙이기
        }
    }, [page]);
    
    return (
        // ... (생략) ...
        <div className="search-results-list">
            {/* 기존처럼 책 카드들 나열하는 map 함수 영역 */}
            {searchResults.map((book) => (
                <div key={book.itemId} className="..."> ... </div>
            ))}
            
            {/* 🔥 화면의 가장 밑바닥에 보이지 않는 센서를 달아둡니다! */}
            <div ref={ref} style={{ height: '50px', backgroundColor: 'transparent' }}>
                {/* 바닥에 닿았을 때 로딩 중 표시 (선택사항) */}
                {isLoading && <p style={{textAlign: 'center'}}>책을 더 불러오는 중입니다...</p>}
            </div>
        </div>
    )
}
